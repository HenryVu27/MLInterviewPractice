def clean_text(text: str) -> str:
    """Lowercase, strip URLs/HTML/special chars, collapse whitespace."""
    text = text.lower()
    text = re.sub(r"http\S+|www\.\S+", "", text)
    text = re.sub(r"<[^>]+>", "", text)
    text = re.sub(r"[^a-z0-9\s]", "", text)

# Near-duplicate detection via hashing (MinHash sketch)
def shingle_hash(text: str, k: int = 3) -> set[str]:
    tokens = text.lower().split()
    return {" ".join(tokens[i:i+k]) for i in range(len(tokens)-k+1)}

def jaccard_sim(a: set, b: set) -> float:
    return len(a & b) / len(a | b) if a | b else 0.0

# Flag near-dupes above threshold
seen = []
to_drop = []
for idx, row in df.iterrows():
    shingles = shingle_hash(row["text"])
    if any(jaccard_sim(shingles, s) > 0.8 for s in seen):
        to_drop.append(idx)
    else:
        seen.append(shingles)
df = df.drop(to_drop)

2. Building a Classification Pipeline

End-to-End Spam Classifier with Pretrained Transformer

import torch
from torch.utils.data import Dataset, DataLoader
from transformers import AutoTokenizer, AutoModelForSequenceClassification
from sklearn.metrics import precision_recall_fscore_support
from sklearn.model_selection import train_test_split

texts = ["Buy cheap meds now!", "Meeting at 3pm tomorrow", "You won $1M!", "Project update attached"]
labels = [1, 0, 1, 0]  # 1=spam, 0=ham
train_texts, val_texts, train_labels, val_labels = train_test_split(texts, labels, test_size=0.25, random_state=42)

tokenizer = AutoTokenizer.from_pretrained("bert-base-uncased")

class SpamDataset(Dataset):
    def __init__(self, texts, labels, tokenizer, max_len=128):
        self.encodings = tokenizer(texts, truncation=True, padding="max_length", max_length=max_len, return_tensors="pt")
        self.labels = torch.tensor(labels)

    def __len__(self):
        return len(self.labels)

    def __getitem__(self, idx):
        item = {k: v[idx] for k, v in self.encodings.items()}
        return item, self.labels[idx]

train_loader = DataLoader(SpamDataset(train_texts, train_labels, tokenizer), batch_size=2, shuffle=True)
val_loader = DataLoader(SpamDataset(val_texts, val_labels, tokenizer), batch_size=2)

model = AutoModelForSequenceClassification.from_pretrained("bert-base-uncased", num_labels=2)
optimizer = torch.optim.AdamW(model.parameters(), lr=2e-5)
loss_fn = torch.nn.CrossEntropyLoss()

for epoch in range(3):
    model.train()
    for batch, lbl in train_loader:
        optimizer.zero_grad()
        out = model(**batch)
        loss = loss_fn(out.logits, lbl)
        loss.backward()
        optimizer.step()

    model.eval()
    all_preds, all_labels = [], []
    with torch.no_grad():
        for batch, lbl in val_loader:
            preds = model(**batch).logits.argmax(dim=-1)
            all_preds.extend(preds.tolist())
            all_labels.extend(lbl.tolist())
    p, r, f1, _ = precision_recall_fscore_support(all_labels, all_preds, average="binary")
    print(f"Epoch {epoch+1}: P={p:.3f} R={r:.3f} F1={f1:.3f}")

3. K-Means Clustering

K-Means from Scratch (NumPy)

import numpy as np

def kmeans(X, k, max_iters=100, seed=42):
    rng = np.random.RandomState(seed)
    n = X.shape[0]

    # Random initialization: pick k random points as centroids
    indices = rng.choice(n, size=k, replace=False)
    centroids = X[indices]

    for _ in range(max_iters):
        # Assign each point to nearest centroid
        dists = np.linalg.norm(X[:, None] - centroids[None], axis=2)  # (n, k)
        assignments = np.argmin(dists, axis=1)

        # Update centroids to mean of assigned points
        new_centroids = np.empty_like(centroids)
        for j in range(k):
            members = X[assignments == j]
            if len(members) > 0:
                new_centroids[j] = members.mean(axis=0)
            else:
                new_centroids[j] = centroids[j]

        if np.allclose(centroids, new_centroids):
            break
        centroids = new_centroids

    return centroids, assignments

X = np.random.randn(200, 5)
centroids, labels = kmeans(X, k=3)
for i in range(3):
    print(f"Cluster {i}: {np.sum(labels == i)} points")

4. Logistic & Linear Regression from Scratch

Logistic Regression (NumPy)

import numpy as np

def sigmoid(z): return 1 / (1 + np.exp(-np.clip(z, -500, 500)))

def logistic_regression(X: np.ndarray, y: np.ndarray, lr: float = 0.1, epochs: int = 1000) -> np.ndarray:
    """Binary logistic regression via gradient descent. Returns weights."""
    n, d = X.shape
    X_b = np.hstack([np.ones((n, 1)), X])  # add bias
    W = np.zeros(d + 1)
    for _ in range(epochs):
        preds = sigmoid(X_b @ W)
        grad = X_b.T @ (preds - y) / n
        W -= lr * grad
    return W

def predict(X: np.ndarray, W: np.ndarray) -> np.ndarray:
    X_b = np.hstack([np.ones((X.shape[0], 1)), X])
    return (sigmoid(X_b @ W) >= 0.5).astype(int)

np.random.seed(42)
X = np.random.randn(200, 3)
y = (X @ [1, -2, 0.5] + np.random.randn(200) * 0.5 > 0).astype(float)
W = logistic_regression(X, y, lr=0.5, epochs=2000)
acc = (predict(X, W) == y).mean()
print(f"Accuracy: {acc:.3f}")
print(f"Gradient formula: dW = X.T @ (sigmoid(X @ W) - y) / n")

Linear Regression (NumPy)

import numpy as np

def linear_regression_closed(X: np.ndarray, y: np.ndarray) -> np.ndarray:
    """Normal equation: W = (X^T X)^{-1} X^T y."""
    X_b = np.hstack([np.ones((X.shape[0], 1)), X])
    return np.linalg.pinv(X_b.T @ X_b) @ X_b.T @ y

def linear_regression_gd(X: np.ndarray, y: np.ndarray, lr: float = 0.01, epochs: int = 1000) -> np.ndarray:
    """Gradient descent: dW = X^T (X W - y) / n."""
    X_b = np.hstack([np.ones((X.shape[0], 1)), X])
    W = np.zeros(X_b.shape[1])
    n = len(y)
    for _ in range(epochs):
        grad = X_b.T @ (X_b @ W - y) / n
        W -= lr * grad
    return W

np.random.seed(42)
X = np.random.randn(100, 2)
y = X @ [3, -1.5] + 2 + np.random.randn(100) * 0.3

W_closed = linear_regression_closed(X, y)
W_gd = linear_regression_gd(X, y, lr=0.1, epochs=2000)
print(f"Closed-form: bias={W_closed[0]:.2f}, weights={W_closed[1:]}")
print(f"GD solution: bias={W_gd[0]:.2f}, weights={W_gd[1:]}")
Interviewers love asking "implement logistic regression from scratch." Know the gradient: dW = X.T @ (predictions - y) / n. For linear regression, know both the normal equation and the GD approach.

Mean-Pooled Sentence Embedding (Proper Way)

import torch
from transformers import AutoModel, AutoTokenizer

def mean_pooling(model_output, attention_mask):
    """Mean pooling ignoring padding tokens."""
    hidden = model_output.last_hidden_state          # (batch, seq_len, dim)
    mask = attention_mask.unsqueeze(-1).float()       # (batch, seq_len, 1)
    mask = mask.expand(hidden.size())                 # (batch, seq_len, dim)
    summed = torch.sum(hidden * mask, dim=1)          # (batch, dim)
    counts = torch.clamp(mask.sum(dim=1), min=1e-9)   # (batch, dim)
    return summed / counts

tokenizer = AutoTokenizer.from_pretrained("bert-base-uncased")
model = AutoModel.from_pretrained("bert-base-uncased")
model.eval()

text: str = "Schedule a meeting with the design team."
tokens = tokenizer(text, return_tensors="pt", padding=True, truncation=True, max_length=128)

with torch.no_grad():
    outputs = model(**tokens)

embedding = mean_pooling(outputs, tokens["attention_mask"])
print(embedding.shape)  # (1, 768)

Batch Embedding with torch.no_grad()

import torch
from transformers import AutoModel, AutoTokenizer

def embed_batch(texts, model, tokenizer, max_length=128, device="cpu"):
    """Embed texts using mean pooling. Returns (N, hidden_dim)."""
    tokens = tokenizer(texts, return_tensors="pt", padding=True, truncation=True, max_length=max_length)
    tokens = tokens.to(device)
    model.to(device)
    model.eval()

    with torch.no_grad():
        outputs = model(**tokens)

    # Mean pooling: average token embeddings, ignoring padding
    hidden = outputs.last_hidden_state                        # (N, seq_len, dim)
    mask = tokens["attention_mask"].unsqueeze(-1).float()      # (N, seq_len, 1)
    mask = mask.expand(hidden.size())                          # (N, seq_len, dim)
    summed = torch.sum(hidden * mask, dim=1)                   # (N, dim)
    counts = torch.clamp(mask.sum(dim=1), min=1e-9)            # (N, dim)
    embeddings = summed / counts

    # L2 normalize
    return torch.nn.functional.normalize(embeddings, p=2, dim=1)

tokenizer = AutoTokenizer.from_pretrained("bert-base-uncased")
model = AutoModel.from_pretrained("bert-base-uncased")

texts = ["Turn off the lights", "Set alarm for 7am", "What is the weather?"]
embs = embed_batch(texts, model, tokenizer)
print(embs.shape)  # (3, 768)

[CLS] vs Mean Pooling vs Max Pooling
[CLS] token -- fast, single vector. Works well when the model was pre-trained with [CLS] as a sentence-level representation (BERT's NSP). Can be noisy for retrieval.
Mean pooling -- average all token embeddings (excluding padding). Generally the best default for sentence embeddings. Used by sentence-transformers.
Max pooling -- take the max across the sequence dimension for each hidden dim. Can capture salient features but loses ordering information. Rarely the best choice.

7. Similarity Search with Embeddings

Cosine Similarity with torch.nn.functional

import torch
import torch.nn.functional as F

def pairwise_cosine_similarity(a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
    """Cosine similarity between all pairs. a: (M, D), b: (N, D) -> (M, N)."""
    a_norm = F.normalize(a, p=2, dim=1)
    b_norm = F.normalize(b, p=2, dim=1)
    return torch.mm(a_norm, b_norm.t())

import numpy as np

Approximate Nearest Neighbor with FAISS (HNSW)

import numpy as np
import faiss

dim: int = 384
num_vectors: int = 100_000
embeddings = np.random.randn(num_vectors, dim).astype("float32")
faiss.normalize_L2(embeddings)

hnsw_index = faiss.IndexHNSWFlat(dim, 32)  # M=32 neighbors per node
hnsw_index.hnsw.efConstruction = 200
hnsw_index.hnsw.efSearch = 64
hnsw_index.metric_type = faiss.METRIC_INNER_PRODUCT

hnsw_index.add(embeddings)

query = np.random.randn(1, dim).astype("float32")
faiss.normalize_L2(query)
distances, indices = hnsw_index.search(query, k=10)
print(f"HNSW top-10: {indices[0]}")
FAISS Index Types -- Trade-offs for On-Device
IndexFlatIP / IndexFlatL2 -- exact search, O(N) per query. Perfect for <100K vectors. No training needed.
IndexIVFFlat -- partitions space into Voronoi cells. Much faster search, slight accuracy loss. Needs training. Good for 100K-10M vectors.
IndexHNSWFlat -- graph-based. Excellent recall, fast search, but high memory (stores full vectors + graph). Good for on-device if memory allows.
IndexIVFPQ -- IVF + product quantization. Compresses vectors heavily (e.g., 384 floats -> 48 bytes). Best for on-device with tight RAM constraints on Galaxy phones.

        self.labels = torch.tensor(labels, dtype=torch.long)

    def __len__(self):
        return len(self.labels)

    def __getitem__(self, idx):
        return self.encodings["input_ids"][idx], self.encodings["attention_mask"][idx], self.labels[idx]

tokenizer = AutoTokenizer.from_pretrained("bert-base-uncased")
dataset = TextDataset(["Set alarm", "Play jazz", "Weather?"], [0, 1, 2], tokenizer)
loader = DataLoader(dataset, batch_size=2, shuffle=True)

Training Loop with Validation and Early Stopping

import torch
import torch.nn as nn

def train_model(model, train_loader, val_loader, num_epochs=10, lr=2e-5, patience=3, device="cuda"):
    model = model.to(device)
    optimizer = torch.optim.AdamW(model.parameters(), lr=lr)
    criterion = nn.CrossEntropyLoss()
    best_val_loss = float("inf")
    patience_counter = 0

    for epoch in range(num_epochs):
        # Train
        model.train()
        train_loss = 0.0
        for input_ids, attention_mask, labels in train_loader:
            input_ids, attention_mask, labels = input_ids.to(device), attention_mask.to(device), labels.to(device)

            optimizer.zero_grad()
            logits = model(input_ids, attention_mask)
            loss = criterion(logits, labels)
            loss.backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            optimizer.step()
            train_loss += loss.item()

        # Validate
        model.eval()
        val_loss, correct, total = 0.0, 0, 0
        with torch.no_grad():
            for input_ids, attention_mask, labels in val_loader:
                input_ids, attention_mask, labels = input_ids.to(device), attention_mask.to(device), labels.to(device)
                logits = model(input_ids, attention_mask)
                val_loss += criterion(logits, labels).item()
                correct += (logits.argmax(-1) == labels).sum().item()
                total += labels.size(0)

        avg_train = train_loss / len(train_loader)
        avg_val = val_loss / len(val_loader)
        print(f"Epoch {epoch+1}: train={avg_train:.4f} val={avg_val:.4f} acc={correct/total:.4f}")

        # Early stopping
        if avg_val < best_val_loss:
            best_val_loss = avg_val
            patience_counter = 0
            torch.save(model.state_dict(), "best_model.pt")
        else:
            patience_counter += 1
            if patience_counter >= patience:
                print(f"Early stopping at epoch {epoch+1}")
                break

    model.load_state_dict(torch.load("best_model.pt"))
    return model
Training loop pattern: zero_grad -> forward -> loss -> backward -> clip_grad -> step. Always validate after each epoch. Early stopping prevents overfitting: track best val loss and stop if it hasn't improved for patience epochs.

9. Two-Stage Retrieval (Bi-Encoder + Cross-Encoder)

Reranking with Cross-Encoder

import torch
from sentence_transformers import CrossEncoder

def retrieve_and_rerank(query, candidates, bi_encoder, cross_encoder, initial_k=20, final_k=5):
    """Two-stage retrieval: bi-encoder recall + cross-encoder reranking."""
    # Stage 1: bi-encoder retrieves top candidates
    query_emb = bi_encoder.encode([query], normalize_embeddings=True)
    doc_embs = bi_encoder.encode(candidates, normalize_embeddings=True)
    scores = (query_emb @ doc_embs.T)[0]
    top_indices = scores.argsort()[::-1][:initial_k]
    shortlist = [candidates[i] for i in top_indices]

    # Stage 2: cross-encoder reranks the shortlist
    pairs = [[query, doc] for doc in shortlist]
    rerank_scores = cross_encoder.predict(pairs)

    # Sort by rerank score descending
    ranked = sorted(zip(shortlist, rerank_scores), key=lambda x: x[1], reverse=True)
    return ranked[:final_k]

from sentence_transformers import SentenceTransformer

bi_encoder = SentenceTransformer("all-MiniLM-L6-v2")
cross_encoder = CrossEncoder("cross-encoder/ms-marco-MiniLM-L-6-v2")

docs = ["User likes jazz", "User likes rock", "User has a cat", "User works at Samsung"]
results = retrieve_and_rerank("What music does the user enjoy?", docs, bi_encoder, cross_encoder, initial_k=3, final_k=2)
for doc, score in results:
    print(f"{score:.4f} | {doc}")
Why Two-Stage Retrieval (Bi-Encoder + Cross-Encoder)
Bi-encoder -- encodes query and documents independently. Fast (precompute doc embeddings, just encode query at search time). But limited: no cross-attention between query and document tokens.
Cross-encoder -- concatenates query + document and processes jointly through the transformer. Much more accurate (full cross-attention), but O(N) per query since you cannot precompute.
Solution: Bi-encoder retrieves top-K candidates (e.g., 100 from 1M). Cross-encoder reranks those K. Best of both worlds: speed + accuracy.

10. Evaluation Metrics

Computing Retrieval Metrics (Recall@K, NDCG)

import torch
import numpy as np

def recall_at_k(retrieved_indices, relevant_indices, k):
    """Recall@K: fraction of relevant items found in top-K."""
    recalls = []
    for i in range(len(relevant_indices)):
        top_k = set(retrieved_indices[i, :k].tolist())
        relevant = set(relevant_indices[i])
        if len(relevant) == 0:
            continue
        hit_count = len(top_k & relevant)
        recalls.append(hit_count / len(relevant))
    return np.mean(recalls)

def ndcg_at_k(retrieved_indices, relevance_scores, k):
    """NDCG@K: normalized discounted cumulative gain."""
    ndcgs = []
    for i in range(len(relevance_scores)):
        # Compute DCG from retrieved order
        dcg = 0.0
        for rank in range(min(k, retrieved_indices.shape[1])):
            doc_id = retrieved_indices[i, rank].item()
            rel = relevance_scores[i].get(doc_id, 0.0)
            dcg += (2**rel - 1) / np.log2(rank + 2)

        # Compute ideal DCG (best possible ordering)
        ideal_rels = sorted(relevance_scores[i].values(), reverse=True)[:k]
        idcg = 0.0
        for j, r in enumerate(ideal_rels):
            idcg += (2**r - 1) / np.log2(j + 2)

        ndcgs.append(dcg / idcg if idcg > 0 else 0.0)
    return np.mean(ndcgs)

retrieved = torch.tensor([[3, 1, 4, 0, 2], [1, 0, 3, 2, 4]])
relevant = [[1, 3], [0, 1]]
print(f"Recall@3: {recall_at_k(retrieved, relevant, k=3):.4f}")

rel_scores = [{1: 2.0, 3: 1.0, 4: 0.5}, {0: 2.0, 1: 1.0}]
print(f"NDCG@3: {ndcg_at_k(retrieved, rel_scores, k=3):.4f}")

Perplexity Computation

import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

def compute_perplexity(model, tokenizer, text: str, device: str = "cpu") -> float:
    """Perplexity of text under a causal LM. Lower = more confident."""
    model.eval()
    model.to(device)
    encodings = tokenizer(text, return_tensors="pt").to(device)

    with torch.no_grad():
        outputs = model(**encodings, labels=encodings["input_ids"])
        neg_log_likelihood = outputs.loss

    return torch.exp(neg_log_likelihood).item()

tokenizer = AutoTokenizer.from_pretrained("gpt2")
model = AutoModelForCausalLM.from_pretrained("gpt2")

text = "Samsung Bixby is a voice assistant that remembers your preferences."
ppl = compute_perplexity(model, tokenizer, text)
print(f"Perplexity: {ppl:.2f}")
Recall@K -- did we find the relevant items in the top K? NDCG -- are relevant items ranked higher? Both are standard for evaluating retrieval/memory systems. Perplexity measures how confident the LM is in its predictions (lower = better).
